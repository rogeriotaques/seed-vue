### Dismissable

When `dismissable` is given, the alert box will display a `close` button which will emit the `dismiss` event when clicked.

export const Dismissable = () => ({
  components: { SAlert },
  template: `
    <div >
      <s-alert
        v-if="alert1"
        color="primary"
        dismissable
        @dismiss="alert1 = false"
      >
        A dismissable alert box
      </s-alert>
      <s-alert
        v-if="alert2"
        color="primary"
        title="A dismissable alert box"
        dismissable
        @dismiss="alert2 = false"
      >
        That also contain a title.
      </s-alert>
      <p >
        <a v-if="!alert1 || !alert2" href="#" @click.prevent="alert1 = alert2 = true" >Click here to show the alerts again.</a>
      </p>
    </div>
  `,
  data() {
    return {
      alert1: true,
      alert2: true
    };
  }
});
